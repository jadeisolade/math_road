N=10000                       %数据总数

G=0                        %满足条件的数据数量
                
P=0.5                         %某一种情况（点名）出现的概率

for n=1:N                     %N次循环
   
    C=0                       %连续多少天，每次循环设定初始为0天
    for m=1:7                 %一到七天循环
        randshu=rand(1)       %把每次产生的随机数赋给randshu，便于以后的判断
        
        if randshu>P          %大于表示满足条件，即点名
              C=C+1
        end                   %end 表示结束if 判断
        if randshu<P
            C=0
        end
         if  C>=3            %C>=3表示连续三天点名，满足条件，所以G+1
             G=G+1
             break          %满足条件之后，break语句跳出for m=1:7 这个循环
         end
    end     
end
     
odd=G/N                       %注意，每一个条件语句都要以end结尾
